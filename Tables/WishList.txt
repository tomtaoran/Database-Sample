CREATE TABLE `WishList` (
  `UserID` int(11) NOT NULL,
  `GameID` int(11) NOT NULL,
  PRIMARY KEY (`UserID`,`GameID`),
  KEY `fk_WishList_GameID_idx` (`GameID`),
  CONSTRAINT `fk_WishList_GameID` FOREIGN KEY (`GameID`) REFERENCES `Game` (`GameID`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `fk_WishList_UserID` FOREIGN KEY (`UserID`) REFERENCES `Users` (`UserID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
