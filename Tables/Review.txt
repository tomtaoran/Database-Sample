CREATE TABLE `Review` (
  `UserID` int(11) NOT NULL,
  `GameID` int(11) NOT NULL,
  `Rating` int(11) DEFAULT NULL,
  PRIMARY KEY (`GameID`,`UserID`),
  KEY `fk_Review_UserID_idx` (`UserID`),
  CONSTRAINT `fk_Review_GameID` FOREIGN KEY (`GameID`) REFERENCES `Game` (`GameID`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `fk_Review_UserID` FOREIGN KEY (`UserID`) REFERENCES `Users` (`UserID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
